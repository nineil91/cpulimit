45static int kproc2proc(kvm_t *kd, struct kinfo_proc *kproc, struct process *proc)

70int get_next_process(struct process_iterator *it, struct process *p) {
71 if (it->i == it->count)
72 {
73 return -1;
74 }
75 if (it->filter->pid != 0 && !it->filter->include_children)
76 {
77 if (get_single_process(it->kd, it->filter->pid, p) != 0)
78 {
79 it->i = it->count = 0;
80 return -1;
81 }
82 it->i = it->count = 1;
83 return 0;
84 }
85 while (it->i < it->count)
86 {
87 struct kinfo_proc *kproc = &(it->procs[it->i]);
88 if (kproc->ki_flag & P_SYSTEM)
89 {
90 // skip system processes
91 it->i++;
92 continue;
93 }
94 if (it->filter->pid != 0 && it->filter->include_children)
95 {
96 kproc2proc(it->kd, kproc, p);
97 it->i++;
98 if (p->pid != it->filter->pid && p->ppid != it->filter->pid)
99 continue;
100 return 0;
101 }
102 else if (it->filter->pid == 0)
103 {
104 kproc2proc(it->kd, kproc, p);
105 it->i++;
106 return 0;
107 }
108 }
109 return -1;
110}
