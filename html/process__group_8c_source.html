43// look for a process with a given name
44// process: the name of the wanted process. it can be an absolute path name to the executable file
45// or just the file name
46// return: pid of the found process, if it is found
47// 0, if it's not found
48// negative pid, if it is found but it's not possible to control it
49int find_process_by_name(const char *process_name)
50{
51 //pid of the target process
52 pid_t pid = -1;
53
54 //process iterator
55 struct process_iterator it;
56 struct process proc;
57 struct process_filter filter;
58 filter.pid = 0;
59 filter.include_children = 0;
60 init_process_iterator(&it, &filter);
61 while (get_next_process(&it, &proc) != -1)
62 {
63 //process found
64 if (strncmp(basename(proc.command), process_name, strlen(process_name))==0 && kill(pid,SIGCONT)==0) {
65 //process is ok!
66 pid = proc.pid;
67 break;
68 }
69 }
70 if (close_process_iterator(&it) != 0) exit(1);
71 if (pid >= 0) {
72 //ok, the process was found
73 return pid;
74 }
75 else {
76 //process not found
77 return 0;
78 }
79}
